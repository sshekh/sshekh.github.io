template <class node>
struct segtree {
  node *tree;
  int h, n, left_most, right_most;
  const node IDENTITY;
  segtree() { tree = 0; }
  ~segtree() { delete[] tree; }
  void build(int size, const int a[]);
  inline void pushdown(int idx) {
    if(idx > 1) pushdown(idx>>1);
    tree[idx].push(tree[idx<<1], tree[(idx<<1)|1]);
  }
  int u, v; // range is [u, v), single element is u
  node query(int root, int left_leaf, int right_leaf);
  node query(int _u, int _v);
  node query(int idx);
  template <class T> void update(int idx, T d);
  template <class T> void range_update(int root, int left_leaf, int right_leaf, T d);
  template <class T> void range_update(int _u, int _v, T d);
  // searches for the last place i, such that merge[0..i] compares < k, requires op<
  int binary_search(node k);
};

#include <bits/stdc++.h>

using namespace std;

const int N = 10004;
int n, m;

struct node {
  int val, sum;
  node() : val(0), sum(0) {}
  node(int _val) : val(_val), sum(_val) {}
  void merge(const node& l, const node& r) {
    sum = l.sum + r.sum;
  }
  void push(node& l, node& r) {}
  void update(int d) {sum = val = d;}
};

int a[N];

int main() {
  segtree<node> sumt;
  cin >> n;
  for(int i = 0; i < n; i++) cin >> a[i];
  sumt.build(n, a);
  cin >> m;
  int x, y, type;
  node res;
  while(m--) {
    cin >> type >> x >> y;
    x--;
    if(type == 1) {
      res = sumt.query(x, y);
      cout << res.sum << "\n";
    } else {
      sumt.update<int>(x, y);
    }
  }

  return 0;
}

template <class node>
void segtree<node>::build(int size, const int a[]) {
  n = size; h = ceil(log2(n));
  left_most = 1<<h; right_most = left_most<<1;
  if(tree) delete[] tree;
  tree = new node[right_most];
  for(int i = 0; i < n; i++) tree[i|(1<<h)] = (node){a[i]};
  for(int i = left_most - 1; i > 0; i--) tree[i].merge(tree[i<<1], tree[(i<<1)|1]);
}
template <class node>
node segtree<node>::query(int root, int left_leaf, int right_leaf) {
  if(u <= left_leaf && right_leaf <= v) return tree[root];
  int mid = (left_leaf + right_leaf)>>1, lc = root<<1, rc = lc|1;
  tree[root].push(tree[lc], tree[rc]);
  node ret;
  if(v <= mid) ret = query(lc, left_leaf, mid);
  else if(u >= mid) ret = query(rc, mid, right_leaf);
  else ret.merge(query(lc, left_leaf, mid), query(rc, mid, right_leaf));
  return ret;
}
template <class node>
node segtree<node>::query(int _u, int _v) {
  u = _u + left_most; v = _v + left_most;
  if(u >= v) return IDENTITY;
  return query(1, left_most, right_most);
}
template <class node>
node segtree<node>::query(int idx) {
  idx |= (1<<h);
  pushdown(idx);
  return tree[idx];
}
template <class node> template <class T>
void segtree<node>::update(int idx, T d) {
  idx |= (1<<h);
  pushdown(idx>>1);
  tree[idx].update(d);
  idx >>= 1;
  while(idx > 0) {
    tree[idx].merge(tree[idx<<1], tree[(idx<<1)|1]);
    idx >>= 1;
  }
}
template <class node> template <class T>
void segtree<node>::range_update(int root, int left_leaf, int right_leaf, T d) {
  if(u <= left_leaf && right_leaf <= v) return tree[root].update(d);
  int mid = (left_leaf + right_leaf)>>1, lc = root<<1, rc = lc|1;
  tree[root].push(tree[lc], tree[rc]);
  if(u < mid) range_update(lc, left_leaf, mid, d);
  if(v > mid) range_update(rc, mid, right_leaf, d);
  tree[root].merge(tree[lc], tree[rc]);
}
template <class node> template <class T>
void segtree<node>::range_update(int _u, int _v, T d) {
  u = _u + left_most; v = _v + left_most;
  if(u < v) range_update(1, left_most, right_most, d);
}
template <class node>
int segtree<node>::binary_search(node k) {
  int root = 1, lc, rc;
  node nd;
  while(root < left_most) {
    lc = root<<1, rc = lc|1;
    tree[root].push(tree[lc], tree[rc]);
    node m;
    m.merge(nd, tree[lc]);
    if(m < k) {
      nd = m; root = rc;
    } else {
      root = lc;
    }
  }
  node ret;
  ret.merge(nd, tree[root]);
  if(ret < k) return root - left_most;
  else return root - left_most - 1;
}
